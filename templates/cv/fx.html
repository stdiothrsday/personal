import javafx.application.Application;
import javafx.scene.Scene;
import javafx.stage.Stage;

/**
 * JavaFX class p5 creates a scene with with change conversions.
 *
 * @author (Yaw Abaaho)
 * @version (11/24/19)
 */
public class P5 extends Application
{
    public void start(Stage stage)
    {
        Scene scene = new Scene(new P5Pane(), 500, 300);
        stage.setTitle("Change with Minimal Number of Coins");
        stage.setScene(scene);
        stage.show();
    }

    public static void main(String[] args)
    {
        launch(args);
    }
}

import javafx.event.ActionEvent;
import javafx.geometry.HPos;
import javafx.geometry.Pos;
import javafx.scene.control.Label;
import javafx.scene.control.TextField;
import javafx.scene.layout.GridPane;
import javafx.scene.text.Font;

/**
 * JavaFX class P5Pane has labels and textfield for change conversion.
 *
 * @author (Yaw Abaaho)
 * @version (11/24/19)
 */
public class P5Pane extends GridPane
{
    private Label quarters;
    private Label dimes;
    private Label nickels;
    private Label pennies;
    private TextField change;

    public P5Pane()
    {
        Font font = new Font(18);

        Label inputLabel = new Label("Change Amount 0 to 99 cents:");
        inputLabel.setFont(font);
        GridPane.setHalignment(inputLabel, HPos.RIGHT);

        Label outputLabel1 = new Label("Quarters:");
        outputLabel1.setFont(font);
        GridPane.setHalignment(outputLabel1, HPos.RIGHT);

        Label outputLabel2 = new Label("Dimes:");
        outputLabel2.setFont(font);
        GridPane.setHalignment(outputLabel2, HPos.RIGHT);

        Label outputLabel3 = new Label("Nickels:");
        outputLabel3.setFont(font);
        GridPane.setHalignment(outputLabel3, HPos.RIGHT);

        Label outputLabel4 = new Label("Pennies:");
        outputLabel4.setFont(font);
        GridPane.setHalignment(outputLabel4, HPos.RIGHT);

        quarters = new Label("---");
        quarters.setFont(font);
        GridPane.setHalignment(quarters, HPos.CENTER);

        dimes = new Label("---");
        dimes.setFont(font);
        GridPane.setHalignment(dimes, HPos.CENTER);

        nickels = new Label("---");
        nickels.setFont(font);
        GridPane.setHalignment(nickels, HPos.CENTER);

        pennies = new Label("---");
        pennies.setFont(font);
        GridPane.setHalignment(pennies, HPos.CENTER);

        change = new TextField();
        change.setFont(font);
        change.setPrefWidth(50);
        change.setAlignment(Pos.CENTER);
        change.setOnAction(this::convertChange);

        setAlignment(Pos.CENTER);
        setHgap(20);
        setVgap(10);
        setStyle("-fx-background-color: yellow");

        add(inputLabel, 0, 0);
        add(change, 1, 0);

        add(outputLabel1, 0, 1);
        add(quarters, 1, 1);

        add(outputLabel2, 0,2);
        add(dimes, 1, 2);

        add(outputLabel3, 0, 3);
        add(nickels, 1, 3);

        add(outputLabel4, 0, 4);
        add(pennies, 1, 4);
    }

    private void convertChange(ActionEvent event)
    {
        int changeAmount = Integer.parseInt(change.getText());
        int quarter = changeAmount/25;

        changeAmount=changeAmount%25;
        int dime = changeAmount/10;

        changeAmount=changeAmount%10;
        int nickel = changeAmount/5;

        changeAmount=changeAmount%5;
        int penny = changeAmount/1;

        quarters.setText(quarter + "");
        dimes.setText(dime + "");
        nickels.setText(nickel + "");
        pennies.setText(penny + "");
    }
}
